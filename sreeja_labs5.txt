2.shell script to find the given number is prime number or not.

 echo "enter a number"
 read num
 i=2
 while [$i -lt $num]
 do
 if [`expr $num %$i` -eq 0]
 then
 echo "$num is not a prime number"
 echo "since division by $i"

 fi
 i=`expr $i+1`
 done
